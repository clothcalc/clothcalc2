def calculate_fabric1(): weight = float(input("Weight (Kg): ")) gsm = float(input("GSM: ")) width = float(input("Width (cm): ")) # Convert width to meters width_meters = width / 100 # Calculate fabric length in meters fabric_length_meters = (weight * 1000) / (gsm * width_meters) # Convert fabric length from meters to yards fabric_length_yards = fabric_length_meters * 1.09361 # Display result print(fabric_length_yards, "Yards") def calculate_fabric2(): weight = float(input("Weight (Kg): ")) gsm = float(input("GSM: ")) width = float(input("Width (inch): ")) # Convert width to meters width_meters = width * 0.0254 # Calculate fabric length in meters fabric_length_meters = (weight * 1000) / (gsm * width_meters) # Convert fabric length from meters to yards fabric_length_yards = fabric_length_meters * 1.09361 # Display result print(fabric_length_yards, "Yards") print("1. kg to yards, fabric width in centimeters") print("2. kg to yards, fabric width in inch") choice = int(input("Enter your choice: ")) if choice == 1: calculate_fabric1() elif choice == 2: calculate_fabric2() else: print("Invalid choice. Please choose either 1 or 2.")
